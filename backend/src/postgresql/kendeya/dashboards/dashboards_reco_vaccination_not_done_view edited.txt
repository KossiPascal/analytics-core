-- @name: dashboards_reco_vaccination_not_done_view
-- @type: matview
-- @depends:
-- @auto_depends: true
-- @indexes:
--   - columns: ["month","year","reco_id"]
--     unique: true
--   - columns: ["id"]
--     method: btree

CREATE MATERIALIZED VIEW IF NOT EXISTS dashboards_reco_vaccination_not_done_view AS

    WITH base_children AS (
        SELECT *
        FROM dash_max_vaccination_view v
        WHERE v.child_id IS NOT NULL
    ),

    children_not_done AS (
        SELECT *
        FROM base_children v
        WHERE
            v.vaccine_VAR_2 IS NOT TRUE
            AND (
                (v.child_age_in_days > 0 AND v.is_birth_vaccine_ok IS NOT TRUE) OR
                (v.child_age_in_days >= 42 AND v.is_six_weeks_vaccine_ok IS NOT TRUE) OR
                (v.child_age_in_days >= 70 AND v.is_ten_weeks_vaccine_ok IS NOT TRUE) OR
                (v.child_age_in_days >= 98 AND v.is_forteen_weeks_vaccine_ok IS NOT TRUE) OR
                (v.child_age_in_months >= 9 AND v.is_nine_months_vaccine_ok IS NOT TRUE) OR
                (v.child_age_in_months >= 15 AND v.is_fifty_months_vaccine_ok IS NOT TRUE)
            )
    ),

    families_grouped AS (
        SELECT
            v.reco_id,
            v.family_id,
            jsonb_build_object(
                'family', jsonb_build_object(
                    'id', v.family_id,
                    'name', v.family_given_name,
                    'fullname', v.family_name,
                    'code', v.family_external_id,
                    'phone', COALESCE(NULLIF(v.householder_phone, ''), NULLIF(v.householder_phone_other, ''))
                ),
                'data', jsonb_agg(
                    jsonb_build_object(
                        'child', jsonb_build_object(
                            'id', v.child_id,
                            'name', v.child_name,
                            'code', v.child_code,
                            'sex', v.child_sex,
                            'age_in_days', v.child_age_in_days,
                            'age_in_months', v.child_age_in_months,
                            'age_in_years', v.child_age_in_years,
                            'age_str', v.child_age_str
                        ),
                        'VAR_2', jsonb_build_object(
                            'done', v.vaccine_VAR_2,
                            'date', v.vaccine_VAR_2_date,
                            'reason', v.no_VAR_2_reason
                        )
                    )
                    ORDER BY v.child_name
                )
            ) AS family_json
        FROM children_not_done v
        GROUP BY v.reco_id, v.family_id, v.family_given_name,
                v.family_name, v.family_external_id,
                v.householder_phone, v.householder_phone_other
    ),

    final_agg AS (
        SELECT
            a.month,
            a.year,
            a.reco_id,
            jsonb_agg(f.family_json ORDER BY f.family_json->'family'->>'name') AS children_vaccines
        FROM year_month_reco_grid_view a
        JOIN families_grouped f ON f.reco_id = a.reco_id
        GROUP BY a.month, a.year, a.reco_id
    )

    SELECT
        CONCAT(month, '-', year, '-', reco_id) AS id,
        month,
        year,
        reco_id,
        children_vaccines
    FROM final_agg
    WHERE children_vaccines IS NOT NULL
    AND jsonb_array_length(children_vaccines) > 0;
